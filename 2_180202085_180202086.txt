    public void zaman() {
        long toplam;
        zaman1 = System.currentTimeMillis();
        toplam = saat * 3600 + dakika * 60 + saniye;
        toplam = (toplam + (zaman1 - zaman) / 1000);
        saat = (int) (toplam / 3600);
        toplam = toplam % 3600;
        dakika = (int) (toplam / 60);
        toplam = toplam % 60;
        saniye = (int) toplam;

        LocalTime time = LocalTime.of(saat, dakika, saniye);

        try {
            Statement sta = baglan.createStatement();
            if (tur.equals("Film")) {

                String sorgu = "Update kullaniciprogram set izleme_suresi='" + time.toString() + "' WHERE program_id=" + program_id + " AND kullanici_id="+giris_id+"";
                sta.execute(sorgu);
            }
            else
            {
                String sorgu = "Update kullaniciprogram set izleme_suresi='" + time.toString() + "' WHERE program_id=" + program_id + " AND kullanici_id="+giris_id+" AND k_bolum='"+bolum_index+"'";
                sta.execute(sorgu);
            }
            
        } catch (Exception e) {
            System.out.println("hata olustu...  " + e);
        }
        saat=0;
        dakika=0;
        saniye=0; 
        baglanti_kes();
       baglan();
    }
